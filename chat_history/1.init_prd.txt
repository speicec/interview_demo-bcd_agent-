  分叉点 1: Agent 的"大脑"用什么？

  ┌──────────────────────────┬──────┬──────┬────────────────────┬────────┐
  │           方案           │ 速度 │ 成本 │       准确度       │ 复杂度 │
  ├──────────────────────────┼──────┼──────┼────────────────────┼────────┤
  │ 纯规则引擎               │ 即时 │ 零   │ 中等（边界情况弱） │ 低     │
  ├──────────────────────────┼──────┼──────┼────────────────────┼────────┤
  │ 纯 LLM (Claude API)      │ 1-3s │ $/次 │ 高（解释自然）     │ 极低   │
  ├──────────────────────────┼──────┼──────┼────────────────────┼────────┤
  │ 混合：规则判断 + LLM解释 │ 1-2s │ 低$  │ 最高               │ 中     │
  └──────────────────────────┴──────┴──────┴────────────────────┴────────┘

  我的直觉是 混合方案
  最适合医疗场景：规则引擎负责"超标/正常/偏低"的硬判断（零幻觉），LLM
  负责生成通俗易懂的解释文本。但这取决于你的容忍度。

  分叉点 2: 架构形态？

  方案 A: 纯前端单文件 HTML
    Browser → Claude API (key 暴露在浏览器)
    构建时间: ~2小时
    风险: API Key 泄露

  方案 B: Next.js / 小后端
    Browser → API Route → Claude API
    构建时间: ~4小时
    风险: 无

  方案 C: Python FastAPI + 模板
    Browser → FastAPI → Claude API + 规则引擎
    构建时间: ~3小时
    好处: Python 医疗库生态

  层: 框架
  选型: React + Vite
  理由: 你指定的，Vite 开发体验最快
  ────────────────────────────────────────
  层: UI

  ────────────────────────────────────────

  理由: 最快的 JSON Schema 校验库
  ────────────────────────────────────────
  层: Schema定义

  理由: 标准化，可扩展
  ────────────────────────────────────────
  层: 规则引擎
  选型: 纯 JS 对象映射

  ────────────────────────────────────────
  层: LLM
  选型: DeepSeek (OpenAI兼容)

  ────────────────────────────────────────
  层: 后端
  选型: Express.js

  ────────────────────────────────────────
  层: 对话记忆
  选型: 内存 Map (session)
  理由: MVP 不需要持久化

  // server/src/agent/session.ts  —— 核心骨架

  import { createAgentSession, defineTool, SessionManager } from
  "@earendil-works/pi-coding-agent";
  import { Type } from "typebox";

  // 1. 自定义工具：规则引擎
  const analyzeBloodReportTool = defineTool({
    name: "analyze_blood_report",
    label: "分析血常规",
    description:
  "基于2023中国成人血常规参考标准分析血常规报告，返回每项指标的判定等级和解读",
    parameters: Type.Object({
      reportJson: Type.String({ description: "血常规报告 JSON 字符串" }),
    }),
    execute: async (_id, params) => {
      const report = JSON.parse(params.reportJson);
      const result = ruleEngine.analyze(report);  // 纯 TS 规则引擎
      return {
        content: [{ type: "text", text: JSON.stringify(result, null, 2) }],
        details: { result },
      };
    },
  });

  // 2. 自定义工具：参考范围查询
  const getReferenceRangeTool = defineTool({
    name: "get_reference_range",
    label: "查询参考范围",
    description: "查询2023指南中特定血常规指标的参考范围及临床意义",
    parameters: Type.Object({
      code: Type.String({ description: "指标编码，如 WBC, HGB, PLT" }),
      gender: Type.Optional(Type.String({ description: "性别: male/female" })),
      age: Type.Optional(Type.Number({ description: "年龄" })),
    }),
    execute: async (_id, params) => {
      const ref = referenceRangeDB.lookup(params.code, params.gender,
  params.age);
      return {
        content: [{ type: "text", text: JSON.stringify(ref) }],
        details: { ref },
      };
    },
  });

  // 3. 创建 Agent 会话
  export async function createChatSession(reportContext: AnalysisResult) {
    const { session } = await createAgentSession({
      model: getModel("deepseek", "deepseek-v4-flash"),
      thinkingLevel: "off",
      authStorage,
      modelRegistry,
      tools: [],                          // 不需要代码工具
      customTools: [analyzeBloodReportTool, getReferenceRangeTool],
      sessionManager: SessionManager.inMemory(),
      resourceLoader: createMedicalResourceLoader(reportContext),
    });

    return session;
  }
